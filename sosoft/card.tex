% Shadow macro with inverse color: #1=shadow color, #2=content
% Shadow is offset 0.35pt down-right in the specified shadow color at 30% opacity
\newcommand{\sh}[2]{%
  \setbox0=\hbox{#2}%
  \rlap{\raisebox{-0.35pt}{\hspace{0.35pt}{\color{#1!30}\copy0}}}%
  \box0%
}
