% Shadow macro — #1 is the inverse/opposite color, rendered at 50% behind text
\newcommand{\sh}[2]{%
  \setbox0=\hbox{#2}%
  \rlap{\raisebox{-0.4pt}{\hspace{0.4pt}{\color{#1!50}\copy0}}}%
  \box0%
}
