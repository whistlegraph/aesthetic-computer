// 3. Set up Powershell Script (script.ps1)

$firstRun = $true;

while ($true) {
    $process = Get-Process "Sunset" -ErrorAction SilentlyContinue
    if (-not $process) {
        if ($firstRun) {
          Start-Process "C:\Program Files (x86)\Sunset\Sunset.exe"
          $firstRun = $false
        } else {
          Restart-computer
          break
        }
    }
    Start-Sleep -Seconds 1
}
